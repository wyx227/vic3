BYN_chairman_clothes_trigger = { #民国总统
	exists = scope:character
	scope:character = {
		is_ruler = yes
		owner = {
			OR = {
				has_law = law_type:law_parliamentary_republic
				has_law = law_type:law_presidential_republic
				has_law = law_type:law_corporate_state
			}
			NOT = {has_law = law_type:law_autocracy}
			NOT = {country_has_primary_culture = cu:manchu}
		}
		culture = {
			AND = {
				has_discrimination_trait = tradition_sinosphere
			    has_discrimination_trait_group = language_group_sinitic
			}
		}
	}
}

BYN_general_clothes_trigger = { #民国陆军将领通用
	exists = scope:character
	scope:character = {
		is_ruler = no
		owner = {
			OR = {
				has_law = law_type:law_parliamentary_republic
				has_law = law_type:law_presidential_republic
				has_law = law_type:law_corporate_state
				has_law = law_type:law_monarchy
			}
			NOT = {has_law = law_type:law_autocracy}
			NOT = {country_has_primary_culture = cu:manchu}
		}
		culture = {
			AND = {
				has_discrimination_trait = tradition_sinosphere
			    has_discrimination_trait_group = language_group_sinitic
			}
		}
		has_role = general
		commander_rank >= 0
	} 
}
