chinese_han_imperial_clothes_trigger = { #明皇帝
	exists = scope:character
	scope:character = {
		is_ruler = yes
		OR = {
			AND = {
				exists = c:CEC
				owner = c:CEC
				owner = {
					OR = {
						has_law = law_type:law_monarchy
						has_law = law_type:law_theocracy
					}
				}
				culture = {
			        AND = {
				        has_discrimination_trait = tradition_sinosphere
			            has_discrimination_trait_group = language_group_sinitic
			        }
		        }
			}
			AND = {
				exists = c:CHI
				owner = c:CHI
				owner = {
					OR = {
						has_variable = jianbian
						NOT = { country_has_primary_culture = cu:manchu }
					}
					has_law = law_type:law_monarchy
				}
			}
		}
	}
}
