je_chinese_dragon = {#龙腾九霄总日志
    icon = "gfx/interface/icons/event_icons/china.dds"

	group = je_group_foreign_affairs

	is_shown_in_lobby = {
		c:CEC ?= this
	}

	is_shown_when_inactive = {
		c:CEC ?= this
	}

	possible = {
		has_modifier = modifier_huanghan
		has_technology_researched = civilizing_mission
		legitimacy >= 80
		custom_tooltip = {
			text = regeneration_complete_tt
			has_variable = regeneration_complete
		}
		custom_tooltip = {
			text = ban_opium_cec_complete_tt
			has_variable = ban_opium_cec_complete
		}
		is_subject = no
		hidden_trigger = {
			NOT = {
				has_variable = je_chinese_dragon_complete
			}
		}
		is_diplomatic_play_committed_participant = no
	}

	status_desc = {
        first_valid = {
            triggered_desc = {
                desc = chinese_dragon_pursuit_tt
                trigger = {
                    has_variable = chinese_dragon_pursuit
                }
            }
            triggered_desc = {
                desc = chinese_dragon_fanaticism_tt
                trigger = {
                    has_variable = chinese_dragon_fanaticism
                }
            }
            triggered_desc = {
                desc = chinese_dragon_abandon_tt
                trigger = {
                	has_variable =  chinese_dragon_abandon
                }
            }
        }
    }

	immediate = {
		trigger_event = { id = chinese_dragon.1 }
		add_journal_entry = { type = je_chinese_dragon_indochina }
        add_journal_entry = { type = je_chinese_dragon_siberia }
        add_journal_entry = { type = je_chinese_dragon_korea }
		add_journal_entry = { type = je_chinese_dragon_indonesia }
        add_journal_entry = { type = je_chinese_dragon_tibet }
        add_journal_entry = { type = je_chinese_dragon_xiyu_1 }
	}

	complete = {
		is_at_war = no
		is_subject = no
		gold_reserves > 0
		government_legitimacy >= 60
		has_global_highest_gdp = yes
		gdp > {
			value = global_gdp
			multiply = 0.5
		}
		hidden_trigger = {
			OR = {
				has_variable = chinese_dragon_pursuit
				has_variable = chinese_dragon_fanaticism
			}
		}

		    custom_tooltip = {
			    text = je_chinese_dragon_siberia_complete_tt
			    has_variable = je_chinese_dragon_siberia_complete
		    }

			custom_tooltip = {
			    text = je_chinese_dragon_indonesia_complete_tt
			    has_variable = je_chinese_dragon_indonesia_complete
		    }
		    custom_tooltip = {
			    text = je_chinese_dragon_tibet_complete_tt
			    has_variable = je_chinese_dragon_tibet_complete
		    }
		    custom_tooltip = {
			    text = je_chinese_dragon_xiyu_1_complete_tt
			    has_variable = je_chinese_dragon_xiyu_1_complete
		    }
		trigger_if = {
			limit = {
				has_variable = chinese_dragon_fanaticism
			}
		    custom_tooltip = {
			    text = je_chinese_dragon_korea_complete_tt
			    has_variable = je_chinese_dragon_korea_complete
		    }
			custom_tooltip = {
			    text = je_chinese_dragon_indochina_complete_tt
			    has_variable = je_chinese_dragon_indochina_complete
			    }
			custom_tooltip = {
			    text = je_chinese_dragon_japan_complete_tt
			    has_variable = je_chinese_dragon_japan_complete
		    }
			custom_tooltip = {
			    text = je_chinese_dragon_india_complete_tt
			    has_variable = je_chinese_dragon_india_complete
		    }
			custom_tooltip = {
			    text = je_chinese_dragon_xiyu_2_complete_tt
			    has_variable = je_chinese_dragon_xiyu_2_complete
		    }
		}
	}

	on_complete = {
		trigger_event = { id = chinese_dragon.2 }
		set_variable = je_chinese_dragon_complete
		every_country = {
			limit = {
				NOT = { country_rank >= rank_value:great_power }
				NOT = { is_country_type = unrecognized }
				NOT = { is_country_type = decentralized }
				NOT = {
					OR = {
							any_scope_treaty = {
								binds = ROOT
								any_scope_article = {
									has_type = alliance
								}
							}
							is_subject_of = ROOT
							this.power_bloc ?= {
								power_bloc_leader = ROOT
							}
							AND = {
								any_scope_culture = {
									limit = { is_primary_culture_of = PREV }
									shares_trait_with_any_primary_culture = ROOT
								}
							}
					}
				}
			}
			set_country_type = unrecognized
			change_relations = {
				country = ROOT
				value = -50
			}
		}
	}

	event_outcome_completed_effect_desc = {
        header = chinese_dragon.2.a
        effect = {
			if = {
			limit = {
				has_variable = chinese_dragon_pursuit
			}
			add_modifier = {
				name = modifier_chinese_dragon_pursuit_completed
#				days = very_long_modifier_time
#				is_decaying = yes
			}
		}
		else_if = {
			limit = {
				has_variable = chinese_dragon_fanaticism
			}
			add_modifier = {
				name = modifier_chinese_dragon_fanaticism_completed
#				days = very_long_modifier_time
#				is_decaying = yes
			}
		}
        }
    }

	timeout = 10950

	on_timeout = {
		trigger_event = { id = chinese_dragon.3 }
		set_variable = je_chinese_dragon_complete
	}

	fail = {
		OR = {
			is_subject = yes
		 	AND = {
				country_rank < rank_value:great_power
				gold_reserves < 0
		 	}
			custom_tooltip = {
			    text = chinese_dragon_abandon_2_tt
			    has_variable = chinese_dragon_abandon
		    }
			custom_tooltip = {
				text = lost_war_tt
				OR = {
					has_variable = recently_lost_war
					AND = {
						has_variable = recently_had_war
						NOT = {
							has_variable = recently_won_war
						}
					}
				}
			}
		}
	}

	on_fail = {
		if = {
			custom_tooltip = {
			    text = chinese_dragon_abandon_2_tt
			    has_variable = chinese_dragon_abandon
		    }
			set_variable = je_chinese_dragon_complete
		}
		else = {
			trigger_event = { id = chinese_dragon.3 }
		    set_variable = je_chinese_dragon_complete
		}	
	}

	event_outcome_timeout_effect_desc = {
        header = chinese_dragon.3.a
        effect = {
			if = {
			limit = {
				is_subject = no
				any_subject_or_below = {
					NOT = { any_overlord_or_above = { NOT = { ROOT = this } } }
					NOT = {
						is_country_type = unrecognized
					}
				}
			}
			every_subject_or_below = {
				limit = {
					NOT = { any_overlord_or_above = { NOT = { ROOT = this } } }
					OR = {
						NOT = {
							is_country_type = unrecognized
						}
						is_subject_type = subject_type_protectorate
						is_subject_type = subject_type_dominion
						is_subject_type = subject_type_puppet
						is_subject_type = subject_type_colony
						is_subject_type = subject_type_chartered_company
					}
				}
				make_independent = yes
			}
		}
		    set_country_type = unrecognized
		    ruler = {
			    kill_character = yes
		    }
			add_modifier = {
				name = modifier_chinese_dragon_abandon_completed
				months = 600
				is_decaying = yes
			}
        }
    }

	on_monthly_pulse = {
		random_events = {
			100 = 0
			10 = chinese_dragon.13
			10 = chinese_dragon.14
			10 = chinese_dragon.15
		}
	}

	weight = 100

	should_be_pinned_by_default = yes
}
